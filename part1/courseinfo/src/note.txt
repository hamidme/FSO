//Advanced solution
function Header({course}){
  return(
    <>
      <h1>{course}</h1>
    </>
  )
}


function Part(props){
  //console.log(props)
  return(
    <>
    <p>
        {props.name} {props.exercises}
      </p>
    </>
  )
}


function Content({name, exercises}){
  return (
    <>
      <Part 
        name = {name}
        exercises = {exercises}
      />
    </>
  )
}

function Total({sum}){
  return(
    <>
      <p>Number of exercises {sum}</p>
    </>
  )
}

function App() {
  const course = 'Half Stack application development'
  const parts = [ 
    {
      CourseName: 'Fundamentals of React',
      exercises: 10
    },

     {
      CourseName: 'Using props to pass data',
      exercises: 7
    },

    {
      CourseName: 'State of a component',
      exercises: 14
    }
  ]

  const contentData = parts.map((item)=>{
      return(
      <Content key={item.CourseName} name={item.CourseName} exercises={item.exercises}/>  
      )
  })

    let sum = 0
    for (let i of parts){
      const exercises = i.exercises
      sum+=exercises
    }
  return (
    <>
      <Header course={course}/>
        {contentData}
      <Total sum = {sum}/>
    </>
  );
}

export default App;
